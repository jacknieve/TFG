Usuario{
Correo String
Password String
Rol int
Id int Unique
FechaRegistro Date
Habilitado Boolean
Bloqueado Boolean
Primary Key(Correo)
}
Mentor{
Correo String
Nombre String
Apellidos String
NivelEstudios String
Telefono String(12)[+00000000000]
Descripcion Text [500 max]
Linkedin String
FechaEliminacion Date
HorasPorMes int
FechaNaz Date
Primary Key(Correo)
Foreing Key(Correo, Usuario(Correo))
}

Mentorizado{
Correo String
Nombre String
Apellidos String
NivelEstudios String
Telefono String(12)[+00000000000]
Descripcion Text [500 max]
Linkedin String
FechaEliminacion Date
FechaNaz Date
Primary Key(Correo)
Foreing Key(Correo, Usuario(Correo))
}

Institucion{
Correo String
Icono String
Nombre String Unique
Color String
PaginaWeb String
Direccion String
Visibilidad Boolean
Primary Key(Correo)
Foreing Key(Correo, Usuario(Correo))
}

SerDeInstitucion{
Mentor String
Institucion String
Puesto String
Primary Key(Mentor, Institucion)
Foreing Key(Mentor, Mentor(Correo))
Foreing Key(Institucion, Institucion(Nombre))
}

AreasMentor{
Mentor String
Nombre String
Primary Key(Mentor, Nombre)
Foreing Key(Mentor, Mentor(Correo))
}

AreasMentorizado{
MentorizadoId String
Nombre String
Primary Key(Mentorizado, Nombre)
Foreing Key(Mentorizado, Mentorizado(Correo))
}

VerificadoPor{
Mentor String
Institucion String
Primary Key(Mentor, Institucion)
Foreing Key(Mentor, Mentor(Correo))
Foreing Key(Institucion, Institucion(Nombre))
}

Mentorizacion{
Mentor String
Mentorizado String
FechaIn date
FechaFin date
Calificacion int
Comentario String
Primary Key(Mentor, Mentorizado, FechaIn)
Foreing Key(Mentor, Mentor(Correo))
Foreing Key(Mentorizado, Mentorizado(Correo))
}

Peticion{
Mentor String
Mentorizado String
Fecha date
Motivo String
Estado String/int
FechaRes Date
Primary Key(Mentor, Mentorizado, Fecha)
Foreing Key(Mentor, Mentor(Correo))
Foreing Key(Mentorizado, Mentorizado(Correo))
}

Extension{
Extension String
Primary Key(Extension)
}

ExtensionesPermitidas{
Extension String
Institucion String
Primary Key(Extension, Institucion)
Foreing Key(Extension, Extension(Extension))
Foreing Key(Institucion, Institucion(Nombre))
}

Notificacion{
Usuario String Index
Id int
Fecha Date
Titulo String
Descripcion Text
Estado String/int
Primary Key(Id)
Foreing Key(Usuario, Usuario(Correo))
}
